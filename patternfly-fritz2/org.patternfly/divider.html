Creates a PatternFly divider component.

A divider is a horizontal separator that creates space or groups items within a page or list.

Samples

import dev.fritz2.dom.html.render
import org.patternfly.DividerVariant.HR
import org.patternfly.DividerVariant.LI
import org.patternfly.divider
import org.patternfly.layout
fun main() { 
   //sampleStart 
   render {
    p { +"First paragraph" }
    divider(HR)
    p { +"Second paragraph" }

    ul {
        li { +"First item" }
        divider(LI)
        li { +"Second item" }
    }

    div(baseClass = "flex".layout()) {
        div { +"First item" }
        divider(vertical = true)
        div { +"Second item" }
    }
} 
   //sampleEnd
}
