Contains functions to easily add notifications from source flows, emitting handlers or listeners.

There are two flavours of functions:

  1. Functions default, error, info, success and warning: Use these functions, if you want to add static notifications and don't need the payload from the source flow.

  2. Function add: Use this function, if you want to use the payload from the source flow to create the notification.

Samples

import dev.fritz2.dom.html.render
import org.patternfly.ButtonVariation
import org.patternfly.ButtonVariation.primary
import org.patternfly.IconPosition
import org.patternfly.Notification
import org.patternfly.buttonIcon
import org.patternfly.clickButton
import org.patternfly.fas
import org.patternfly.icon
import org.patternfly.linkButton
import org.patternfly.pushButton
import org.patternfly.util
fun main() { 
   //sampleStart 
   render {
    div {
        clickButton(primary) {
            +"Click me"
        } handledBy Notification.info("Score!")
    }
} 
   //sampleEnd
}
import dev.fritz2.dom.html.render
import org.patternfly.ChipGroupStore
import org.patternfly.Notification
import org.patternfly.chipGroup
import org.patternfly.chips
import org.patternfly.chip
fun main() { 
   //sampleStart 
   render {
    chipGroup<String> {
        +"Remove one"
        chips("Foo", "Bar")
        store.removes handledBy Notification.add { chip ->
            info("You removed $chip.")
        }
    }
} 
   //sampleEnd
}
