accordionDl

fun RenderContext.accordionDl(singleExpand: Boolean = true, fixed: Boolean = false, id: String? = null, baseClass: String? = null, content: Accordion<HTMLDListElement>.() -> Unit = {}): Accordion<HTMLDListElement>
Content copied to clipboard

Creates an Accordion component based on a <dl> element with nested <dt> and <dd> elements.

Samples

import dev.fritz2.dom.html.render
import org.patternfly.accordionContent
import org.patternfly.accordionDiv
import org.patternfly.accordionDl
import org.patternfly.accordionItem
import org.patternfly.accordionTitle
fun main() { 
   //sampleStart 
   render {
    accordionDl {
        accordionItem {
            accordionTitle { +"Item one" }
            accordionContent {
                p { +"Lorem ipsum dolor sit amet." }
            }
        }
        accordionItem(expanded = true) {
            accordionTitle { +"Item two" }
            accordionContent {
                p { +"Phasellus pretium est a porttitor vehicula." }
            }
        }
        accordionItem {
            accordionTitle { +"Item three" }
            accordionContent {
                p { +"Quisque vel commodo urna." }
            }
        }
    }
} 
   //sampleEnd
}

Parameters

singleExpand

whether only one AccordionContent can be expanded at a time

fixed

whether AccordionContent uses a fixed height

id

the ID of the element

baseClass

optional CSS class that should be applied to the element

content

a lambda expression for setting up the component itself
