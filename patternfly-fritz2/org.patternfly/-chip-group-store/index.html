ChipGroupStore

class ChipGroupStore<T>(idProvider: IdProvider<T, String>) : RootStore<List<T>> , WithIdProvider<T>
Content copied to clipboard

Store containing a list of items used by the ChipGroup component. Each item is identified by the specified IdProvider which defaults to

{ Id.build(it.toString()) }

Samples

import dev.fritz2.dom.html.render
import org.patternfly.ChipGroupStore
import org.patternfly.Notification
import org.patternfly.chipGroup
import org.patternfly.chips
import org.patternfly.chip
fun main() { 
   //sampleStart 
   render {
    data class Demo(val id: String, val name: String)

    val store = ChipGroupStore<Demo> { it.id }
    chipGroup(store) {
        +"Store demo"
        display { demo ->
            chip { +demo.name }
        }
    }

    store.addAll(
        listOf(
            Demo("foo", "Foo"),
            Demo("bar", "Bar")
        )
    )
} 
   //sampleEnd
}
