import org.patternfly.classes
import org.patternfly.component
import org.patternfly.modifier
import kotlin.random.Random
fun main() { 
   //sampleStart 
   val disabled = Random.nextBoolean()
val classes = classes {
    +"button".component()
    +"plain".modifier()
    +("disabled".modifier() `when` disabled)
} 
   //sampleEnd
}
