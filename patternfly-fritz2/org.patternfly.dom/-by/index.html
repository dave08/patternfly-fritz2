By

interface By
Content copied to clipboard

Typesafe CSS selector API.

Use the methods in this class to create arbitrary complex CSS selectors:

#main [data-list-item=foo] a[href^="http://"] .fas.fa-check, .external[hidden]

Samples

import org.patternfly.dom.AttributeOperator
import org.patternfly.dom.By
fun main() { 
   //sampleStart 
   By.group(
    By.id("main")
        .desc(
            By.data("listItem", "foo")
                .desc(
                    By.element("a")
                        .and(By.attribute("href", "https://", AttributeOperator.STARTS_WITH))
                        .child(By.classname("fas", "fa-check"))
                )
        ),
    By.classname("external").and(By.attribute("hidden"))
) 
   //sampleEnd
}
